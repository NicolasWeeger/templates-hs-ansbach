% =====================================================================
%  Conference Poster Template
%  Card-based layout with header band, two columns, footer
%
%  Format : A0 portrait (841 mm x 1189 mm)
%  Engine : pdflatex / lualatex
%
%  HOW TO USE
%  ----------
%  1. Replace [PLACEHOLDERS] with your content (search for "REPLACE")
%  2. Drop your logo into the same folder, update \includegraphics path
%  3. Add references to references.bib
%  4. Adjust accent colors in the palette section if you have a CI
%  5. Compile: pdflatex -> bibtex -> pdflatex -> pdflatex
%
%  CARD TYPES
%  ----------
%  cardblock        primary section (blue header)
%  cardblockaccent  highlight section (teal header)
%  cardblockmuted   low-emphasis section like References (light header)
% =====================================================================

% --- Color palette (REPLACE with your CI colors if available) ---
\definecolor{accentBlue}  {RGB}{ 11, 64,115}   % deep blue, primary
\definecolor{accentTeal}  {RGB}{  0,138,162}   % teal accent
\definecolor{accentLight} {RGB}{233,240,247}   % background tint

\definecolor{accentMuted} {RGB}{110,120,135}   % captions, footer

% --- Beamer theme tweaks ---
\usetheme{default}
\setbeamercolor{background canvas}{bg=posterBg}
\setbeamercolor{itemize item}{fg=accentTeal}
\setbeamercolor{itemize subitem}{fg=accentTeal}

\setbeamertemplate{navigation symbols}{}
\setbeamertemplate{caption}[numbered]
\setbeamercolor{caption name}{fg=accentBlue}

% --- Layout sizes ---
\newlength{\colw}
\setlength{\colw}{0.475\paperwidth}
\setlength{\parskip}{0.4em}

% --- Requirement-tag pill (like a GitHub label) ---
\newcommand{\reqtag}[1]{%
  \tcbox[on line,
         colback=accentLight,colframe=accentBlue!25,
         boxsep=1pt,left=4pt,right=4pt,top=1pt,bottom=1pt,
         arc=2pt,boxrule=0.3pt]{%
    \footnotesize\textcolor{accentBlue}{\textbf{#1}}}%
}

% --- Placeholder helpers (TikZ-drawn; replace with \includegraphics) ---
% Used so the template compiles without external assets. Swap each
% \placeholderLogo / \placeholderFigure call for \includegraphics{...}.
\newcommand{\placeholderFigure}[1][0.98\linewidth]{%
  \begin{tikzpicture}
    \node[draw=cardEdge,fill=accentLight,minimum width=#1,
          minimum height=10cm,align=center,font=\Large\bfseries,
          text=accentMuted]{FIGURE PLACEHOLDER\\[0.4em]
          \mdseries\normalsize replace with \texttt{\textbackslash includegraphics\{...\}}};
  \end{tikzpicture}%
}

% --- Header / footer color bands ---
\newcommand{\posterHeader}{%
  \begin{tikzpicture}[remember picture,overlay]
    \fill[accentBlue] ([yshift=0pt]current page.north west)
        rectangle ([yshift=-13cm]current page.north east);
    \fill[accentTeal] ([yshift=-13cm]current page.north west)
        rectangle ([yshift=-13.4cm]current page.north east);
  \end{tikzpicture}%
}

\newcommand{\posterFooter}{%
  \begin{tikzpicture}[remember picture,overlay]
    \fill[accentBlue]  ([yshift=4.5cm]current page.south west)
        rectangle ([yshift=0cm]current page.south east);
    \fill[accentTeal]  ([yshift=4.5cm]current page.south west)
        rectangle ([yshift=4.7cm]current page.south east);
  \end{tikzpicture}%
}
